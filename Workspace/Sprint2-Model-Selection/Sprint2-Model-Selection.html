glm_data_proc <- function(data.table) {
    data.table <- data.table %>%
        rowwise() %>%
        mutate(ma_mean = mean(ma1, ma2, ma3, ma4, ma5, ma6)) %>%
        select(-c(ma1, ma2, ma3, ma4, ma5, ma6))
    return(data.table)
}
